CELL 3: Start vLLM server with OpenAI API
-------------------------------------------
import subprocess
import time
import requests

# Start vLLM in background
proc = subprocess.Popen([
    "python", "-m", "vllm.entrypoints.openai.api_server",
    "--model", "suckunalol/pub-ai-merged",
    "--port", "8000",
    "--trust-remote-code",
    "--gpu-memory-utilization", "0.9",
    "--dtype", "auto"
], stdout=subprocess.PIPE, stderr=subprocess.PIPE)

print("[vLLM] Starting model load... (2-3 minutes)")
time.sleep(30)

# Check if server is ready
for i in range(60):
    try:
        resp = requests.get("http://localhost:8000/v1/models", timeout=5)
        if resp.status_code == 200:
            print("[vLLM] Server ready!")
            break
    except:
        pass
    print(f"[vLLM] Waiting... {i+1}/60")
    time.sleep(5)


CELL 4: Expose via ngrok
------------------------
from pyngrok import ngrok

# Create public URL
public_url = ngrok.connect(8000, "http")
print(f"\n[SUCCESS] API Endpoint: {public_url}/v1/chat/completions")
print(f"\nUpdate your Pub-AI .env:")
print(f"  VLLM_API_URL={public_url}/v1")
print(f"  AI_PROVIDER=openai")

# Keep tunnel alive
ngrok_process = ngrok.get_ngrok_process()
ngrok_process.proc.wait()
